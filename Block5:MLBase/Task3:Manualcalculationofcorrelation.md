# Рассчитайте линейную корреляцию Пирсона, на основе данных.
## Какой вывод можно сделать на основе полученного результата? Можно ли утверждать, что существует причинно-следственная связь между количеством чашек кофе, выпитых студентами в течение экзаменационного дня, и их итоговым баллом за экзамен?

## Кофе: 1, 1, 2, 2, 2, 2, 3, 3, 3, 4
## Баллы: 85, 88, 79, 81, 84, 65, 67, 58, 76, 49

Среднее кофе = (1+1+2+2+2+2+3+3+3+4) / 10 = 23/10 = 2,3
Средний балл = (85+88+79+81+84+65+67+58+76+49) / 10 = 732/10 = 73,2

## Считаем сумму произведений отклонений:

(1-2,3)(85-73,2) = (-1,3)(11,8) = -15,34
(1-2,3)(88-73,2) = (-1,3)(14,8) = -19,24
(2-2,3)(79-73,2) = (-0,3)(5,8) = -1,74
(2-2,3)(81-73,2) = (-0,3)(7,8) = -2,34
(2-2,3)(84-73,2) = (-0,3)(10,8) = -3,24
(2-2,3)(65-73,2) = (-0,3)(-8,2) = 2,46
(3-2,3)(67-73,2) = (0,7)(-6,2) = -4,34
(3-2,3)(58-73,2) = (0,7)(-15,2) = -10,64
(3-2,3)(76-73,2) = (0,7)(2,8) = 1,96
(4-2,3)(49-73,2) = (1,7)(-24,2) = -41,14
Сумма = -93,6

## Считаем сумму квадратов отклонений для кофе:

(-1,3)^2 + (-1,3)^2 + (-0,3)^2 + (-0,3)^2 + (-0,3)^2 + (-0,3)^2 + (0,7)^2 + (0,7)^2 + (0,7)^2 + (1,7)^2
= 1,69 + 1,69 + 0,09 + 0,09 + 0,09 + 0,09 + 0,49 + 0,49 + 0,49 + 2,89 = 8,2

## Считаем сумму квадратов отклонений для баллов:

(11,8)^2 + (14,8)^2 + (5,8)^2 + (7,8)^2 + (10,8)^2 + (-8,2)^2 + (-6,2)^2 + (-15,2)^2 + (2,8)^2 + (-24,2)^2
= 139,24 + 219,04 + 33,64 + 60,84 + 116,64 + 67,24 + 38,44 + 231,04 + 7,84 + 585,64 = 1499,6

## Вычисляем коэффициент корреляции:
r = -93,6 / √(8,2 × 1499,6)
r = -93,6 / √12296,72
r = -93,6 / 110,89
r = -0,84
Ответ: -0,84
